============================================================
 RmikuOS shell 重定向 + 管道 测试集
 用法:在 shell 里逐条敲(或整块粘贴),对照「预期」检查。
 假设当前在 /tmp(可写)。开头先 cd /tmp。
============================================================

cd /tmp

------------------------------------------------------------
 组 A:基础重定向(覆盖 / 追加 / 输入)
------------------------------------------------------------

# A1: > 覆盖,创建新文件
echo hello > fa
cat fa
# 预期: hello

# A2: > 覆盖已存在文件(截断,不残留)
echo xxxxxxxxxx > fa
echo bb > fa
cat fa
# 预期: bb        (不是 bbxxxxxxxx)

# A3: >> 追加
echo aaa > fb
echo bbb >> fb
cat fb
# 预期两行:
#   aaa
#   bbb

# A4: < 输入重定向(配合一个会读 stdin 的命令,比如 grep)
echo findme > fc
grep find < fc
# 预期: findme    (grep 从 fc 读到 stdin,匹配 find)

# A5: > 创建到不存在的文件(O_CREAT)
echo new > fnew
cat fnew
# 预期: new

------------------------------------------------------------
 组 B:紧贴(无空格)的重定向符
------------------------------------------------------------

# B1: > 紧贴前后
echo hi>fd
cat fd
# 预期: hi

# B2: >> 紧贴
echo aa>>fe
echo bb>>fe
cat fe
# 预期两行: aa / bb

# B3: < 紧贴
echo data > ff
grep dat<ff
# 预期: data

------------------------------------------------------------
 组 C:单管道
------------------------------------------------------------

# C1: 基础管道
echo hello | grep hell
# 预期: hello

# C2: 管道无匹配
echo hello | grep zzz
# 预期: (空,grep 没匹配)

# C3: ls 接 grep(确认 ls 输出能进管道)
ls | grep fa
# 预期: 至少列出 fa(以及其他 fa 开头的)

------------------------------------------------------------
 组 D:多级管道(新能力)
------------------------------------------------------------

# D1: 三级管道
echo abc | grep a | grep b
# 预期: abc       (a 和 b 都匹配)

# D2: 三级,中间过滤掉
echo abc | grep a | grep z
# 预期: (空)

# D3: 四级管道(若 MAX_SEGMENTS 够)
echo hello | grep h | grep e | grep l
# 预期: hello

------------------------------------------------------------
 组 E:管道 + 重定向 结合(核心新能力)
------------------------------------------------------------

# E1: 管道末尾接 > 输出到文件
echo hello | grep h > ge
cat ge
# 预期: hello

# E2: < 输入 + 管道
echo findthis > gf
grep find < gf | grep this
# 预期: findthis

# E3: < 输入 + 管道 + > 输出(全都有!)
echo target > gg
grep tar < gg | grep get > gh
cat gh
# 预期: target

# E4: 管道末尾 >> 追加
echo one | grep o >> gi
echo two | grep t >> gi
cat gi
# 预期两行: one / two

# E5: 多级管道 + 两端重定向
echo pipeline < gg | grep p | grep i > gj
cat gj
# 预期: pipeline   (注意:gg 内容被 < 重定向给第一段 stdin,
#                   但 echo 不读 stdin,所以实际 echo 输出 pipeline)
#   ※ 这条用来测「第一段有 < 同时又是管道源」不会崩

------------------------------------------------------------
 组 F:畸形语法 → 必须报错(syntax error),且整行不执行
------------------------------------------------------------

# F1: > 后面没文件
echo zzz >
# 预期: syntax error in redirection

# F2: >> 后面没文件
echo zzz >>
# 预期: syntax error

# F3: >>> (三个 >)
echo zzz >>> a
# 预期: syntax error

# F4: >< 相连
echo zzz >< a
# 预期: syntax error

# F5: <> 相连
echo zzz <> a
# 预期: syntax error

# F6: 管道中间空命令
echo a | | grep b
# 预期: syntax error: empty command

# F7: 管道开头空
| grep b
# 预期: syntax error: empty command

# F8: 验证报错后没有副作用(F1~F7 跑完后,不该产生奇怪文件)
ls
# 预期: 只看到前面正常创建的 fa fb fc ... ge gf ...,
#       没有名为 a 之类畸形测试残留的文件

------------------------------------------------------------
 组 G:引号 与 重定向/管道符 的交互
------------------------------------------------------------

# G1: 引号里的 > 不是重定向
echo "a > b"
# 预期: a > b      (整个字符串,没有重定向发生)

# G2: 引号里的 | 不是管道
echo "x|y"
# 预期: x|y

# G3: 引号里的 < 不是输入重定向
echo "less < than"
# 预期: less < than

# G4: 引号外重定向 + 引号内容
echo "hello world" > gq
cat gq
# 预期: hello world

# G5: 单引号
echo 'a > b | c'
# 预期: a > b | c   (单引号内全是字面)

------------------------------------------------------------
 组 H:同向多重定向(后者覆盖前者,和 bash 一致)
------------------------------------------------------------

# H1: 两个 >,内容进最后一个
echo zzz > hb > hc
cat hc
# 预期: zzz
cat hb
# 预期: (空,hb 被创建但没写)

# H2: >> 然后 >(最后是覆盖)
echo first > hd
echo second >> hd > hd
cat hd
# 预期: second   (最后一个是 > 覆盖,所以截断后写 second)
#   ※ 若你的语义是「最后生效=覆盖」则 hd 只有 second

------------------------------------------------------------
 组 I:内建命令不受影响(分流正确)
------------------------------------------------------------

# I1: cd/pwd 正常(无管道无重定向,走内建)
cd /tmp
pwd
# 预期: /tmp

# I2: 内建命令后接普通命令
mkdir testdir
ls | grep testdir
# 预期: testdir/

# I3: 清理
rm -r testdir
ls | grep testdir
# 预期: (空)

------------------------------------------------------------
 测试完成。清理(可选):
------------------------------------------------------------
rm fa fb fc fd fe ff fnew ge gf gg gh gi gj gq hb hc hd
